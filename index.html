3D Sphere Visualization with Plotly

- plotly - numpy import numpy as np import plotly.graph_objects as go # Tạo dữ liệu hình cầu theta = np.linspace(0, 2 * np.pi, 100) phi = np.linspace(0, np.pi, 100) theta, phi = np.meshgrid(theta, phi) # Tọa độ hình cầu r = 3 # bán kính X = r * np.sin(phi) * np.cos(theta) Y = r * np.sin(phi) * np.sin(theta) Z = r * np.cos(phi) # Vẽ hình cầu 3D bằng Plotly fig = go.Figure(data=[go.Surface(z=Z, x=X, y=Y, colorscale='Viridis')]) # Hiển thị hình cầu fig.show()
